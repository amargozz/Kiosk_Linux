[########## IDENTIFY DISK AND PARTITIONS CREATION (Boot, SWAP, /) ##########]
fdisk -l
fdisk /dev/sda      # volume name
g                   # create GPT table
n  1  Enter  +2G    # /sda1 Boot EFI System Partition
  t  1              # EFI systam
n  2  Enter  +8G    # /sda2 SWAP
  t  2  19          # Linux swap
n  3  Enter  Enter  # /sda3 /
w

[########## FORMATING ##########]
mkfs.fat -F32 /dev/sda1
mkswap /dev/sda2
swapon /dev/sda2
mkfs.ext4 /dev/sda3

mount /dev/sda3 /mnt
mkdir /mnt/boot
mount /dev/sda1 /mnt/boot

[########## ARCH INSTALLATION ##########]
  (With no modification in mirrorlist)
pacstrap -K /mnt base linux linux-firmware
genfstab -U /mnt >> /mnt/etc/fstab

arch-chroot /mnt
pacman -S --noconfirm nano sudo
ln -sf /usr/share/zoneinfo/America/El_Salvador /etc/localtime
hwclock --systohc

echo "LANG=en_US.UTF-8" > /etc/locale.conf
